406 def get_GITT_initial_concs(currents,run_times, c0, R, a, L, electrode_charge):
407  F = 96485.3321 #faraday constant
408  #volume fraction of active material
409  e_act = (a*R)/3
410 
411  #array of initial concentrations, using the fact that
412  # C(T=t) = C0 + ((i_app*t)/(F*e_act*L))
413 
414  # if the electrode charge is positive, then a negative current corresponds
415  # to gaining concentration, and so concentration should rise.
416  # We correct for this by flipping the sign here.
417  currents = np.array(currents)
418  if electrode_charge == 'p':
419  currents = -currents
420  return [c0 + (currents[i]*np.sum(run_times[0:i]))/(F*e_act*L) for i in range(len(currents))]

425  '''!@brief Execution of SPM solver in parallel to run a GITT test over nprocs cores.
426  @details Execution of SPM solver in parallel to run a test experiment on a battery

428  W. Weppner and R. A. Huggins 1977 J. Electrochem. Soc. 124 1569. Due to the equilibration time between each current
429  step applied in this technique, it is possible to pre-compute the initial constant concentration in each of the single
430  particles in the model by considering the amount of lithium removed in each current step.
431  Errors from execution are read in and further execution prevented if necessary.
432  @param[in] filename: Name of user input file, no file extension. Note that a version of this file is created for
433  each current step applied in the GITT test.

439  currents, run_times, wait_times.

445  @param[in] dt: Timestep size, float > 0.
446  @param[in] c0: Initial concentration, float >= 0.
447  @param[in] D: Diffusion constant, float.
448  @param[in] R: Width of block, float > 0.
449  @param[in] a: Particle surface area per unit volume, float >= 0.
450  @param[in] L: Electrode thickness, float >= 0.
451  '''
452  #first, generate the arrays for initial concentration
453  #unpack params
454  [dt, c0, D, R, a, L,electrode_charge] = params
455
