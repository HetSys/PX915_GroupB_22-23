282 def first_order_sensitivities(Q, variables, Q0, eps=1e-6):
283  x0 = np.array([v.mean() for v in variables])
284  dQ_dx = np.zeros(len(x0))
285  for i, xi in enumerate(x0):
286  xp = x0.copy()
287  h = eps*xi
288  xp[i] = xi + h
289  dQ_dx[i] = (Q(xp) - Q0)/h
290  return dQ_dx

292 def second_order_sensitivities(Q, variables, Q0, eps=1e-4):
293  x0 = np.array([v.mean() for v in variables])
294  dQ2_dx2 = np.zeros((len(x0), len(x0)))
295 
296  for i, xi in enumerate(x0):
297  for j, xj in enumerate(x0):
298  if i == j:
299  # diagonal matrix entries
300  hi = eps*xi
301 
302  xp = x0.copy()
303  xp[i] = xi + hi
304 
305  xm = x0.copy()
306  xm[i] = xi - hi
307 
308  dQ2_dx2[i,i] = (Q(xp) - 2*Q0 + Q(xm))/hi**2
309 
310  elif i > j:
311  # off-diagonal matrix entries, noting symmetry on swapping of i and j
312  hi = eps*xi
313  hj = eps*xj
314 
315  x1 = x0.copy()
316  x1[i] += hi
317  x1[j] += hj
318 
319  x2 = x0.copy()
320  x2[i] += hi
321  x2[j] -= hj
322 
323  x3 = x0.copy()
324  x3[i] -= hi
325  x3[j] += hj
326 
327  x4 = x0.copy()
328  x4[i] -= hi
329  x4[j] -= hj
330 
331  dQ2_dx2[i,j] = (Q(x1) - Q(x2) - Q(x3) + Q(x4))/(4*hi*hj)
332 
333  # fill upper triangle
334  i_upper = np.triu_indices(len(x0))
335  dQ2_dx2[i_upper] = dQ2_dx2.T[i_upper]
336 
337  return dQ2_dx2
