104  def first_order_sensitivities(Q, variables, Q0, eps=1e-6):
105  x0 = np.array([v.mean() for v in variables])
106  dQ_dx = np.zeros(len(x0))
107  for i, xi in enumerate(x0):
108  xp = x0.copy()
109  h = eps*xi
110  xp[i] = xi + h
111  dQ_dx[i] = (Q(xp) - Q0)/h
112  return dQ_dx

114  def second_order_sensitivities(Q, variables, Q0, eps=1e-4):
115  x0 = np.array([v.mean() for v in variables])
116  dQ2_dx2 = np.zeros((len(x0), len(x0)))
117 
118  for i, xi in enumerate(x0):
119  for j, xj in enumerate(x0):
120  if i == j:
121  # diagonal matrix entries
122  hi = eps*xi
123 
124  xp = x0.copy()
125  xp[i] = xi + hi
126 
127  xm = x0.copy()
128  xm[i] = xi - hi
129 
130  dQ2_dx2[i,i] = (Q(xp) - 2*Q0 + Q(xm))/hi**2
131 
132  elif i > j:
133  # off-diagonal matrix entries, noting symmetry on swapping of i and j
134  hi = eps*xi
135  hj = eps*xj
136 
137  x1 = x0.copy()
138  x1[i] += hi
139  x1[j] += hj
140 
141  x2 = x0.copy()
142  x2[i] += hi
143  x2[j] -= hj
144 
145  x3 = x0.copy()
146  x3[i] -= hi
147  x3[j] += hj
148 
149  x4 = x0.copy()
150  x4[i] -= hi
151  x4[j] -= hj
152 
153  dQ2_dx2[i,j] = (Q(x1) - Q(x2) - Q(x3) + Q(x4))/(4*hi*hj)
154 
155  # fill upper triangle
156  i_upper = np.triu_indices(len(x0))
157  dQ2_dx2[i_upper] = dQ2_dx2.T[i_upper]
158 
159  return dQ2_dx2
