444 def get_GITT_initial_concs(currents,run_times, c0, R, a, L, electrode_charge):
445  '''@brief Calculates the initial concentrations for each step in a multi-step parallelised GITT test.
446 
447  @details Computes initial flat concentrations for the initial step of a parallelised multi-step GITT test.
448 
449  The formula used for this calculation is:
450  C(T=t) = C_0 + \frac{i_app * t}{F*e_act * L}, where C_0 is C(T=0).
451 
452  As current is defined as positive for charging, the sign of the current vector must be flipped when considering
453  the positive electrode (cathode) which will discharge during charging.
454 
455  @param[in] currents A vector containing current values to apply at each current step, must be floats and have the same length as
456  start_times, run_times, wait_times.
457  @param[in] run_times A vector containing the run time of each current step, must be floats and have the same length as
458  start_times, currents, wait_times.
459  @param[in] c0 Initial concentration, must be a float greater than or equal to 0.
460  @param[in] R Radius of the sphere, must be a float greater than 0.
461  @param[in] a Particle surface area per unit volume, must be a float greater than or equal to 0.
462  @param[in] L Electrode thickness, must be a float greater than or equal to 0.
463  @param[in] electrode_charge Electrode charge, mus be a single character 'p' or 'n'.
464  '''
465  F = 96485.3321 #faraday constant
466  #volume fraction of active material
467  e_act = (a*R)/3
468 
469  #array of initial concentrations, using the fact that
470  # C(T=t) = C0 + ((i_app*t)/(F*e_act*L))
471 
472  # if the electrode charge is positive, then a negative current corresponds
473  # to gaining concentration, and so concentration should rise.
474  # We correct for this by flipping the sign here.
475  currents = np.array(currents)
476  if electrode_charge == 'p':
477  currents = -currents
478  return [c0 + (currents[i]*np.sum(run_times[0:i]))/(F*e_act*L) for i in range(len(currents))]

655 def optimise_parallelism(n_procs,n_gitt_steps=1,full_battery=False):
656  '''!@brief Decides the correct number of threads to use for a given simulation.
657 
658  @details Given a user supplied number of processors, as well as further details about the simulation
659  (the number of current steps if it's a GITT test, if the simulation is a full cell or not), this function
660  decides on the optimum number of threads to use. It never picks a number of threads greater than 4,
661  as any larger than this and the communication overhead between threads during the matrix solve begins to dominate
662  and the computation time increases. This function therefore allows a balanced optimum parallelisation strategy.
663  @param[in] n_procs: The number of processors that the user is happy for the program to exploit. The program will exploit a number
664  of cores less than or equal to this number. Integer.
665  @param[in] n_gitt_steps: The number of GITT steps if the experiment being performed is a GITT test. 1 by default (in the case of
666  the experiment not being a GITT test) integer.
667  @param[in] full_battery: Bool, True means that both sides of a battery are being simulated, whilst False means only one side.
668 
669  The function works by:
670  1. Deciding what number of threads is best to use given the type of simulation and input parameters, up to a maximum of 4.
671  Note that this is based off the principle that parallelisation over independent seperate instances of the solver is always better
672  than multithreading.
673  2. Setting the environment variable.
674  '''
675  #function which decides how many MKL_NUM_THREADS to use
676  # based on: total number of processors available (n_procs)
677  # number of nodes in each matrix inversion (node_num)
678  # number of steps in a gitt test if it's being run.
679 
680  # essentially, compute n_gitt_steps*2(if full battery)
681  # look at this number. If it is <n_procs/4, use 4 thread, if it is
682  # n_procs/4< but <n_procs/2, use 2 thread, if it is >n_procs/2 use 1 thread
683 
684  if full_battery == True:
685  multip = 2
686  else:
687  multip = 1
688 
689  n_gitt_full_batt = multip*n_gitt_steps
690 
691  if n_gitt_full_batt<=int(n_procs/4):
692  os.environ['MKL_NUM_THREADS']='4'
693  os.system('echo set MKL_NUM_THREADS to $MKL_NUM_THREADS')
694  elif (n_gitt_full_batt>int(n_procs/4) and n_gitt_full_batt<=int(n_procs/2)):
695  os.environ['MKL_NUM_THREADS']='2'
696  os.system('echo set MKL_NUM_THREADS to $MKL_NUM_THREADS')
697  else:
698  os.environ['MKL_NUM_THREADS']='1'
